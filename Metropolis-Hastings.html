n = 20000
x0 = 2 # last x
xs = c()
mhq = function(x) {
    return(rnorm(1, mean=x, sd=2))
}
mhdq = function(x1, x2) {
    return(dnorm(x1, mean=x2, sd=2))
}
for (i in seq(0, 1, length=n)) {
    q = mhq(x0)
    A = min(1, (ptilde(q)*mhdq(x0,q))/(ptilde(x0)*mhdq(q, x0)))
    u = runif(1, min=0, max=1)
    if (u <= A) {
        x0 = q
    }
    xs = append(xs, x0)
}
